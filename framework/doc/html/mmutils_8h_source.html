140static inline char mm_nibble_to_hex_char(uint8_t nibble)
141{
142 nibble &= 0x0f;
143 if (nibble < 0x0a)
144 {
145 return '0' + nibble;
146 }
147 else
148 {
149 return 'A' + nibble - 0x0a;
150 }
151}
