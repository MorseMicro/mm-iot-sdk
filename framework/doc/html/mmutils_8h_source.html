134static inline char mm_nibble_to_hex_char(uint8_t nibble)
135{
136 nibble &= 0x0f;
137 if (nibble < 0x0a)
138 {
139 return '0' + nibble;
140 }
141 else
142 {
143 return 'A' + nibble - 0x0a;
144 }
145}
