91static const char *result_code_to_string(enum test_result result)
92{
93 switch (result)
94 {
95 case TEST_NO_RESULT: return "";
96 case TEST_PASSED: return F_GREEN("PASS");
97 case TEST_SKIPPED: return F_BLUE("SKIP");
98 case TEST_FAILED: return F_RED("FAIL");
99 case TEST_FAILED_NON_CRITICAL: return F_YELLOW("FAIL");
100 }
101 MMOSAL_ASSERT(false);
102}

112static void run_test_steps(const struct test_step * const steps[], size_t num_steps,
113 struct test_counters *ctrs)
114{
115 static char log_buf[1024];
116
117 size_t ii;
118 for (ii = 0; ii < num_steps; ii++)
119 {
120 const struct test_step *step = steps[ii];
121 size_t log_buf_len = sizeof(log_buf);
122 log_buf[0] = '\0';
123
124 LOG_PRINTF(F_BOLD("%-60s "), step->description);
125 LOG_FLUSH();
126
127 enum test_result result = step->exec(log_buf, log_buf_len);
128 if (result != TEST_NO_RESULT)
129 {
130 LOG_PRINTF("[ %s ]", result_code_to_string(result));
131 }
132 LOG_WRITE("\n");
133
134 if (log_buf[0] != '\0')
135 {
136 LOG_WRITE("\n");
137 LOG_WRITE(log_buf);
138 }
139
140 switch (result)
141 {
142 case TEST_NO_RESULT: ctrs->no_result++; break;
143 case TEST_PASSED: ctrs->pass++; break;
144 case TEST_SKIPPED: break;
145 case TEST_FAILED: ctrs->fail++; break;
146 case TEST_FAILED_NON_CRITICAL: ctrs->fail++; break;
147 }
148
149 if (result == TEST_FAILED)
150 {
151 break;
152 }
153 }
154}

160void app_init(void)
161{
162 /* Having deep sleep enabled can complicate debugging. Given the purpose of this applications
163 * is to validate that you can communicate to the MM-Chip it will be disabled by default. */
164 mmhal_set_deep_sleep_veto(MMHAL_VETO_ID_APP_MIN);
165
166 struct test_counters ctrs = { 0 };
167 unsigned num_tests = sizeof(test_steps)/sizeof(test_steps[0]);
168
169 LOG_WRITE(F_BOLD("\n\nMM-IoT-SDK Porting Assistant\n"));
170 LOG_WRITE("----------------------------\n\n");
171 run_test_steps(test_steps, num_tests, &ctrs);
172
173
174 LOG_PRINTF("\n\n%u total test steps. %u passed, %u failed, %u no result, %u skipped\n",
175 num_tests, ctrs.pass, ctrs.fail, ctrs.no_result,
176 num_tests - ctrs.no_result - ctrs.pass - ctrs.fail);
177}
