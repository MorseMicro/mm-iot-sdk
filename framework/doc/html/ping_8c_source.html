102void app_init(void)
103{
105 struct mmping_args args = MMPING_ARGS_DEFAULT;
106
107 set_debug_state(DEBUG_STATE_BOOTING_CHIP);
108
109 printf("\n\nMorse Ping Demo (Built " __DATE__ " " __TIME__ ")\n\n");
110
111 /* Initialize and connect to Wi-Fi, blocks till connected */
112 app_wlan_init();
113
114 set_debug_state(DEBUG_STATE_CONNECTING);
115
116 app_wlan_start();
117
118 set_debug_state(DEBUG_STATE_CONNECTED);
119
120 /* Delay to allow communications to settle so we measure only idle current */
121 mmosal_task_sleep(150);
122
123 set_debug_state(DEBUG_STATE_CONNECTED_IDLE);
124
125 mmosal_task_sleep(1000);
126
127 set_debug_state(DEBUG_STATE_PINGING);
128
129 /* Get the target IP */
130 strncpy(args.ping_target, PING_REMOTE_IP, sizeof(args.ping_target));
131 mmconfig_read_string("ping.target", args.ping_target, sizeof(args.ping_target));
132
133 enum mmipal_status status = mmipal_get_local_addr(args.ping_src, args.ping_target);
134 if (status != MMIPAL_SUCCESS)
135 {
136 printf("Failed to get local address for PING\n");
137 }
138 args.ping_count = PING_COUNT;
139 mmconfig_read_uint32("ping.count", &args.ping_count);
140
141 args.ping_size = PING_DATA_SIZE;
142 mmconfig_read_uint32("ping.size", &args.ping_size);
143
144 args.ping_interval_ms = PING_INTERVAL_MS;
145 mmconfig_read_uint32("ping.interval", &args.ping_interval_ms);
146
147 mmping_start(&args);
148 printf("\nPing %s %lu(%lu) bytes of data.\n", args.ping_target, args.ping_size,
149 (MMPING_ICMP_ECHO_HDR_LEN + args.ping_size));
150 mmosal_task_sleep(args.ping_interval_ms);
151
152 struct mmping_stats stats;
153 uint32_t next_update_time_ms = mmosal_get_time_ms() + UPDATE_INTERVAL_MS;
154 unsigned last_ping_recv_count = 0;
155 mmping_stats(&stats);
156 while (stats.ping_is_running)
157 {
158 mmosal_task_sleep(10);
159 mmping_stats(&stats);
160 if (stats.ping_recv_count != last_ping_recv_count ||
161 mmosal_time_has_passed(next_update_time_ms))
162 {
163 printf("(%s) packets transmitted/received = %lu/%lu, "
164 "round-trip min/avg/max = %lu/%lu/%lu ms\n",
165 stats.ping_receiver, stats.ping_total_count, stats.ping_recv_count,
166 stats.ping_min_time_ms, stats.ping_avg_time_ms, stats.ping_max_time_ms);
167 last_ping_recv_count = stats.ping_recv_count;
168 next_update_time_ms = mmosal_get_time_ms() + UPDATE_INTERVAL_MS;
169 }
170 }
171
172 set_debug_state(DEBUG_STATE_PING_DONE);
173
174 uint32_t loss = 0;
175 if (stats.ping_total_count == 0)
176 {
177 loss = 0;
178 }
179 else
180 {
181 loss = (1000 * (stats.ping_total_count - stats.ping_recv_count) * 100 /
182 stats.ping_total_count);
183 }
184 printf("\n--- %s ping statistics ---\n%lu packets transmitted, %lu packets received, ",
185 stats.ping_receiver, stats.ping_total_count, stats.ping_recv_count);
186 printf("%lu.%03lu%% packet loss\nround-trip min/avg/max = %lu/%lu/%lu ms\n",
187 loss/1000, loss%1000, stats.ping_min_time_ms, stats.ping_avg_time_ms,
188 stats.ping_max_time_ms);
189
190 /* Delay to allow communications to settle so we measure only idle current */
191 mmosal_task_sleep(150);
192
193 set_debug_state(DEBUG_STATE_IDLE);
194
195 uint32_t post_ping_delay_ms = POST_PING_DELAY_MS;
196 mmconfig_read_uint32("ping.post_ping_delay_ms", &post_ping_delay_ms);
197 mmosal_task_sleep(post_ping_delay_ms);
198
199 set_debug_state(DEBUG_STATE_TERMINATING);
200
201 /* Disconnect from Wi-Fi */
202 app_wlan_stop();
203}
