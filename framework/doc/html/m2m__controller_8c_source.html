218static bool wlan_connect(struct mmagic_controller *controller)
219{
220 printf("\n\n#### Example WLAN Connect using MMAGIC Controller ####\n\n");
221 enum mmagic_status status;
222
223 printf("Attempting to connect to %s with passphrase %s\n", STRINGIFY(SSID),
224 STRINGIFY(SAE_PASSPHRASE));
225 printf("This may take some time (~30 seconds)\n");
226 status = mmagic_controller_set_wlan_ssid(controller, STRINGIFY(SSID));
227 if (status != MMAGIC_STATUS_OK)
228 {
229 printf("Error %d setting the wlan ssid\n", status);
230 return false;
231 }
232
233 mmagic_controller_set_wlan_password(controller, STRINGIFY(SAE_PASSPHRASE));
234 if (status != MMAGIC_STATUS_OK)
235 {
236 printf("Error %d setting the wlan password\n", status);
237 return false;
238 }
239
240 struct mmagic_core_wlan_connect_cmd_args connect_args = {
241 .timeout = 120000,
242 };
243 status = mmagic_controller_wlan_connect(controller, &connect_args);
244 if (status != MMAGIC_STATUS_OK)
245 {
246 printf("Error %d after sending the connect command\n", status);
247 return false;
248 }
249
250 struct mmagic_core_ip_status_rsp_args ip_status_rsp_args = {};
251 uint32_t timeout = mmosal_get_time_ms() + LINK_STATE_TIMEOUT_MS;
252 while (mmosal_time_lt(mmosal_get_time_ms(), timeout))
253 {
254 status = mmagic_controller_ip_status(controller, &ip_status_rsp_args);
255 if ((status == MMAGIC_STATUS_OK) &&
256 (ip_status_rsp_args.status.link_state == MMAGIC_IP_LINK_STATE_UP))
257 {
258 printf("Link Up\n");
259 printf("Link is up %s. Time: %lu ms, ",
260 ip_status_rsp_args.status.dhcp_enabled ? "(DHCP)" : "(Static)",
261 mmosal_get_time_ms());
262 printf("IP: %s, ", ip_status_rsp_args.status.ip_addr.addr);
263 printf("Netmask: %s, ", ip_status_rsp_args.status.netmask.addr);
264 printf("Gateway: %s", ip_status_rsp_args.status.gateway.addr);
265 printf("\n");
266 return true;
267 }
268 mmosal_task_sleep(500);
269 }
270
271 return false;
272}

291void beacon_monitor_rx_handler(
292 const struct mmagic_wlan_beacon_rx_event_args *args, void *arg)
293{
294 uint32_t offset = 0;
295
296 MM_UNUSED(arg);
297
298 while (offset < args->vendor_ies.len)
299 {
300 uint8_t ie_type;
301 uint8_t ie_len;
302 uint32_t ii;
303
304 if (offset + 2 > args->vendor_ies.len)
305 {
306 printf("Beacon IEs malformed!\n");
307 break;
308 }
309
310 ie_type = args->vendor_ies.data[offset];
311 ie_len = args->vendor_ies.data[offset + 1];
312
313 offset += 2;
314
315 if (offset + ie_len > args->vendor_ies.len)
316 {
317 printf("Beacon IEs malformed!\n");
318 break;
319 }
320
321 printf(" IE type 0x%02x, IE len 0x%02x, Contents: ", ie_type, ie_len);
322 for (ii = 0; ii < ie_len; ii++)
323 {
324 printf("%02x", args->vendor_ies.data[offset++]);
325 }
326 printf("\n");
327 }
328}

369static void tcp_client_example(struct mmagic_controller *controller)
370{
371 printf("\n\n#### Example TCP Client using MMAGIC Controller ####\n\n");
372 enum mmagic_status status;
373
374 struct mmagic_core_tcp_connect_cmd_args tcp_connect_args = {

376 .port = 80

379 status = mmagic_controller_tcp_connect(controller, &tcp_connect_args, &tcp_rsp_args);
380 if (status != MMAGIC_STATUS_OK)

382 printf("Error %d establishing TCP connection\n", status);

386 printf("Opened TCP socket on stream_id %u\n", tcp_stream_id);
387

389 .stream_id = tcp_stream_id,
390 .buffer = {.data = "GET /\n", .len = strlen("GET /\n")}
391 };
392 status = mmagic_controller_tcp_send(controller, &tcp_send_cmd_args);
393 if (status != MMAGIC_STATUS_OK)
394 {
395 printf("Error %d whilst sending data\n", status);
396 MMOSAL_ASSERT(false);
397 }
398 printf("Successfully sent GET request\n");
399
400 struct mmagic_core_tcp_recv_cmd_args tcp_recv_cmd_args = {
401 .stream_id = tcp_stream_id,
402 .len = 1536,
403 .timeout = 1000,
404 };
405 struct mmagic_core_tcp_recv_rsp_args tcp_recv_rsp_args = {};
406 status = mmagic_controller_tcp_recv(controller, &tcp_recv_cmd_args, &tcp_recv_rsp_args);
407 if (status != MMAGIC_STATUS_OK)
408 {
409 printf("Error %d whilst receiving data\n", status);
410 }
411 else
412 {
413 printf("Received: %.12s, Length %u\n", tcp_recv_rsp_args.buffer.data,
414 tcp_recv_rsp_args.buffer.len);
415 }
416
417 struct mmagic_core_tcp_close_cmd_args tcp_close_cmd_args = {.stream_id = tcp_stream_id};
418 status = mmagic_controller_tcp_close(controller, &tcp_close_cmd_args);
419 if (status != MMAGIC_STATUS_OK)
420 {
421 printf("Error %d whilst closing tcp socket\n", status);
422 MMOSAL_ASSERT(false);
423 }
424 printf("Closed TCP Socket\n");

443static void tcp_echo_server_task(void *args)
444{
445 enum mmagic_status status;
446 struct tcp_echo_server_thread_args *thread_args =
447 (struct tcp_echo_server_thread_args *)args;
448 struct mmagic_controller *controller = thread_args->controller;
449 uint8_t stream_id = thread_args->stream_id;
450
451 printf("Accepted a TCP connection on stream_id %u\n", stream_id);
452
453 struct mmagic_core_tcp_recv_cmd_args tcp_recv_cmd_args = {
454 .stream_id = stream_id,
455 .len = 1536,
456 .timeout = 1000,
457 };
458 struct mmagic_core_tcp_recv_rsp_args tcp_recv_rsp_args = {};
459 struct mmagic_core_tcp_send_cmd_args tcp_send_cmd_args =
460 {
461 .stream_id = stream_id
462 };
463
464 while (true)
465 {
466 status = mmagic_controller_tcp_recv(controller, &tcp_recv_cmd_args, &tcp_recv_rsp_args);
467 if (status == MMAGIC_STATUS_TIMEOUT)
468 {
469 continue;
470 }
471 else if (status == MMAGIC_STATUS_CLOSED)
472 {
473 printf("Connection closed by the other side!\n");
474 break;
475 }
476 else if (status != MMAGIC_STATUS_OK)
477 {
478 printf("Error %d whilst receiving data\n", status);
479 break;
480 }
481 printf("Received: %d bytes, echoing.\n", tcp_recv_rsp_args.buffer.len);
482
483 memcpy(tcp_send_cmd_args.buffer.data, tcp_recv_rsp_args.buffer.data,
484 tcp_recv_rsp_args.buffer.len);
485 tcp_send_cmd_args.buffer.len = tcp_recv_rsp_args.buffer.len;
486 status = mmagic_controller_tcp_send(controller, &tcp_send_cmd_args);
487 if (status != MMAGIC_STATUS_OK)
488 {
489 printf("Error %d whilst sending data\n", status);
490 break;
491 }
492 }
493
494 struct mmagic_core_tcp_close_cmd_args tcp_close_cmd_args =
495 {
496 .stream_id = stream_id
497 };
498 status = mmagic_controller_tcp_close(controller, &tcp_close_cmd_args);
499 if (status != MMAGIC_STATUS_OK)
500 {
501 printf("Error %u whilst trying to close the TCP socket on stream_id %u\n",
502 status, stream_id);
503 }
504 printf("Closed TCP socket\n");
505}

513static void tcp_echo_server_start(struct mmagic_controller *controller)
514{
515 printf("\n\n#### Example TCP Echo Server using MMAGIC Controller ####\n\n");
516 enum mmagic_status status;
517 static struct tcp_echo_server_thread_args thread_args;
518
519 struct mmagic_core_tcp_bind_cmd_args tcp_bind_cmd_args = {.port = TCP_ECHCO_SERVER_PORT};
520 struct mmagic_core_tcp_bind_rsp_args tcp_bind_rsp_args = {};
521 status = mmagic_controller_tcp_bind(controller, &tcp_bind_cmd_args, &tcp_bind_rsp_args);
522 if (status != MMAGIC_STATUS_OK)
523 {
524 printf("Error %u whilst opening the tcp socket\n", status);
525 MMOSAL_ASSERT(false);
526 }
527 uint8_t tcp_socket_stream_id = tcp_bind_rsp_args.stream_id;
528 printf("Opened listening socket (Port %u) on stream_id %u\n",
529 tcp_bind_cmd_args.port, tcp_socket_stream_id);
530
531 struct mmagic_core_tcp_accept_cmd_args tcp_accept_cmd_args = {
532 .stream_id = tcp_socket_stream_id
533 };
534 struct mmagic_core_tcp_accept_rsp_args tcp_accept_rsp_args = {};
535 while (true)
536 {
537 status = mmagic_controller_tcp_accept(controller,
538 &tcp_accept_cmd_args,
539 &tcp_accept_rsp_args);
540 if (status != MMAGIC_STATUS_OK)
541 {
542 printf("Error %u whilst trying to accept a TCP connetion\n", status);
543 break;
544 }
545
546 thread_args.stream_id = tcp_accept_rsp_args.stream_id;
547 thread_args.controller = controller;
548
549 mmosal_task_create(tcp_echo_server_task, &thread_args,
550 MMOSAL_TASK_PRI_LOW, 2048, "tcp_echo_server_task");
551 }
552
553 struct mmagic_core_tcp_close_cmd_args tcp_close_cmd_args =
554 {
555 .stream_id = tcp_socket_stream_id
556 };
557 status = mmagic_controller_tcp_close(controller, &tcp_close_cmd_args);
558 if (status != MMAGIC_STATUS_OK)
559 {
560 printf("Error %u whilst trying to close the listening socket on stream_id %u\n",
561 status, tcp_socket_stream_id);
562 }
563 printf("Closed listening socket\n");
564}

571static void run_examples_task(void *args)
572{
573 MM_UNUSED(args);
574
575 agent_started_semb = mmosal_semb_create("agent_started");
576 struct mmagic_controller_init_args init_args = MMAGIC_CONTROLLER_ARGS_INIT;
577 init_args.agent_start_cb = agent_start_handler;
578 init_args.agent_start_arg = (void *)agent_started_semb;
579
580 struct mmagic_controller *controller = mmagic_controller_init(&init_args);
581
582 printf("M2M Controller enabled. Awaiting Agent start\n");
583 mmosal_semb_wait(agent_started_semb, UINT32_MAX);
584
585 if (!wlan_connect(controller))
586 {
587 printf("Failed to connect\n");
588 return;
589 }
590 printf("WLAN connection established\n");
591
592 beacon_monitor_example_start(controller);
593
594 tcp_client_example(controller);
595
596 tcp_echo_server_start(controller);
597}

603void app_init(void)
604{
605 printf("\n\nM2M Controller Example (Built " __DATE__ " " __TIME__ ")\n\n");
606
607 /* Keep controller awake for now */
608 mmhal_set_deep_sleep_veto(MMHAL_VETO_ID_APP_MIN);
609
610 /* Run the examples in a different task to cater for the higher stack size requirements */
611 mmosal_task_create(run_examples_task, NULL,
612 MMOSAL_TASK_PRI_LOW, 2048, "run_examples_task");
613}
