22TEST_STEP(test_step_os_realloc, "Memory reallocation")
23{
24 enum constants
25 {
26 FIRST_ALLOCATION_SIZE = 100,
27 REALLOCATION_SIZE = 200,
28 };
29
30 uint8_t *buf = (uint8_t *)mmosal_malloc(FIRST_ALLOCATION_SIZE);
31 if (buf == NULL)
32 {
33 TEST_LOG_APPEND(
34 "Failed to allocate %d bytes. Check that your heap is configured correctly\n\n",
35 FIRST_ALLOCATION_SIZE);
36 return TEST_FAILED_NON_CRITICAL;
37 }
38 memset(buf, 0xc0, FIRST_ALLOCATION_SIZE);
39
40 uint8_t *buf1 = (uint8_t *)mmosal_realloc(buf, REALLOCATION_SIZE);
41 if (buf1 == NULL)
42 {
43 TEST_LOG_APPEND(
44 "Failed to reallocate %d bytes. Check that your heap supports realloc\n\n",
45 REALLOCATION_SIZE);
46 return TEST_FAILED_NON_CRITICAL;
47 }
48
49 /* Verify the reallocated memory contains the contents of the original block. */
50 unsigned ii;
51 for (ii = 0; ii < FIRST_ALLOCATION_SIZE; ii++)
52 {
53 if (buf1[ii] != 0xc0)
54 {
55 TEST_LOG_APPEND(
56 "Reallocated block contents mismatch at offset %u\n\n", ii);
57 return TEST_FAILED_NON_CRITICAL;
58 }
59 }
60
61 mmosal_free(buf1);
62 return TEST_PASSED;
63}

84enum task_state
85{
86 TASK_NOT_STARTED,
87 TASK_STARTED,
88 TASK_WAITING_FOR_NOTIFICATION,
89 TASK_TERMINATING,
90 TASK_ERROR_WAITING_FOR_NOTIFICATION,
91 TASK_ERROR_GET_ACTIVE_INVALID,
92};
93
95static volatile enum task_state task_state = TASK_NOT_STARTED;
97static struct mmosal_task * volatile task_handle;
98
100static void new_task_main(void *arg)
101{
102 (void)arg;
103
104 task_state = TASK_STARTED;
105
106 /* Sleep for 10 ms; this should yield the task. */
107 mmosal_task_sleep(10);
108
109 /* Verify mmosal_task_get_active() returns the correct task handle. */
110 if (mmosal_task_get_active() != task_handle)
111 {
112 task_state = TASK_ERROR_GET_ACTIVE_INVALID;
113 return;
114 }
115
116 /* Wait for notification. */
117 task_state = TASK_WAITING_FOR_NOTIFICATION;
118 bool ok = mmosal_task_wait_for_notification(UINT32_MAX);
119 if (!ok)
120 {
121 task_state = TASK_ERROR_WAITING_FOR_NOTIFICATION;
122 return;
123 }
124
125 task_state = TASK_TERMINATING;
126
127 /* Delete self */
128 mmosal_task_delete(NULL);
129}

131TEST_STEP(test_step_os_task_creation, "Task creation and preemption")
132{
133 task_handle = mmosal_task_create(
134 new_task_main, NULL, MMOSAL_TASK_PRI_HIGH, 512, "Test Task");
135 if (task_handle == NULL)
136 {
137 TEST_LOG_APPEND(
138 "mmosal_task_create() returned NULL; expected a task handle.\n\n");
139 return TEST_FAILED_NON_CRITICAL;
140 }
141
142 /*
143 * The newly created task should be higher priority. Therefore the following code should
144 * not run until after the task yields.
145 */
146 if (task_state != TASK_STARTED)
147 {
148 TEST_LOG_APPEND(
149 "The task created with mmosal_task_create() did not run.\n\n");
150 return TEST_FAILED_NON_CRITICAL;
151 }
152
153 /* Allow some time for the task to wake up and then wait for notification. */
154 mmosal_task_sleep(50);
155
156 switch (task_state)
157 {
158 case TASK_WAITING_FOR_NOTIFICATION:
159 break;
160
161 case TASK_ERROR_GET_ACTIVE_INVALID:
162 TEST_LOG_APPEND(
163 "mmosal_task_get_active() did not return the correct task handle.\n\n");
164 return TEST_FAILED_NON_CRITICAL;
165
166 case TASK_ERROR_WAITING_FOR_NOTIFICATION:
167 TEST_LOG_APPEND(
168 "mmosal_task_wait_for_notification() unexpectedly returned false.\n\n");
169 return TEST_FAILED_NON_CRITICAL;
170
171 default:
172 TEST_LOG_APPEND(
173 "Task in unexpected state %d.\n\n", task_state);
174 return TEST_FAILED_NON_CRITICAL;
175 }
176
177 /* Notify the task and ensure that it proceeded to the next state */
178 mmosal_task_notify(task_handle);
179
180 switch (task_state)
181 {
182 case TASK_TERMINATING:
183 return TEST_PASSED;
184
185 case TASK_ERROR_GET_ACTIVE_INVALID:
186 TEST_LOG_APPEND(
187 "mmosal_task_get_active() did not return the correct task handle.\n\n");
188 return TEST_FAILED_NON_CRITICAL;
189
190 case TASK_ERROR_WAITING_FOR_NOTIFICATION:
191 TEST_LOG_APPEND(
192 "mmosal_task_wait_for_notification() unexpectedly returned false.\n\n");
193 return TEST_FAILED_NON_CRITICAL;
194
195 default:
196 TEST_LOG_APPEND(
197 "Task in unexpected state %d.\n\n", task_state);
198 return TEST_FAILED_NON_CRITICAL;
199 }
