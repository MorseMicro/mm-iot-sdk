84static uint32_t format_bytes(uint64_t bytes, uint8_t *unit_index)
85{
86 MMOSAL_ASSERT(unit_index != NULL);
87 *unit_index = 0;
88
89 while (bytes >= 1000 && *unit_index < 4)
90 {
91 bytes /= 1000;
92 (*unit_index)++;
93 }
94
95 return bytes;
96}

105static void iperf_report_handler(const struct mmiperf_report *report, void *arg,
106 mmiperf_handle_t handle)
107{
108 (void)arg;
109 (void)handle;
110
111 uint8_t bytes_transferred_unit_index = 0;
112 uint32_t bytes_transferred_formatted = format_bytes(report->bytes_transferred,
113 &bytes_transferred_unit_index);
114
115 printf("\nIperf Report\n");
116 printf(" Remote Address: %s:%d\n", report->remote_addr, report->remote_port);
117 printf(" Local Address: %s:%d\n", report->local_addr, report->local_port);
118 printf(" Transferred: %lu %cBytes, duration: %lu ms, bandwidth: %lu kbps\n",
119 bytes_transferred_formatted, units[bytes_transferred_unit_index],
120 report->duration_ms, report->bandwidth_kbitpsec);
121 printf("\n");
122
123 if ((report->report_type == MMIPERF_UDP_DONE_SERVER) ||
124 (report->report_type == MMIPERF_TCP_DONE_SERVER))
125 {
126 printf("Waiting for client to connect...\n");
127 }
128}

131static void start_tcp_client(void)
132{
133 uint32_t server_port = 0;
134 struct mmiperf_client_args args = MMIPERF_CLIENT_ARGS_DEFAULT;
135
136 /* Get the Server IP */
137 strncpy(args.server_addr, IPERF_SERVER_IP, sizeof(args.server_addr));
138 mmconfig_read_string("iperf.server", args.server_addr, sizeof(args.server_addr));
139
140 mmconfig_read_uint32("iperf.port", &server_port);
141 MMOSAL_ASSERT(server_port <= UINT16_MAX);
142 args.server_port = server_port;
143
144 int amount = IPERF_TIME_AMOUNT;
145 (void)mmconfig_read_int("iperf.amount", &amount);
146 args.amount = amount;
147 if (args.amount < 0)
148 {
149 args.amount *= 100;
150 }
151 args.report_fn = iperf_report_handler;
152
153 mmiperf_start_tcp_client(&args);
154 printf("\nIperf TCP client started, waiting for completion...\n");
155}

184static void start_tcp_server(void)
185{
186 struct mmiperf_server_args args = MMIPERF_SERVER_ARGS_DEFAULT;
187
188 uint32_t local_port = IPERF_SERVER_PORT;
189 mmconfig_read_uint32("iperf.port", &local_port);
190 args.local_port = (uint16_t) local_port;
191
192 args.report_fn = iperf_report_handler;
193
194 mmiperf_handle_t iperf_handle = mmiperf_start_tcp_server(&args);
195 if (iperf_handle == NULL)
196 {
197 printf("Failed to get local address\n");
198 return;
199 }
200 printf("\nIperf TCP server started, waiting for client to connect...\n");
201 struct mmipal_ip_config ip_config;
202 enum mmipal_status status;
203 status = mmipal_get_ip_config(&ip_config);
204 if (status == MMIPAL_SUCCESS)
205 {
206 printf("Execute cmd on AP 'iperf -c %s -p %u -i 1' for IPv4\n",
207 ip_config.ip_addr, args.local_port);
208 }
209
210 struct mmipal_ip6_config ip6_config;
211 status = mmipal_get_ip6_config(&ip6_config);
212 if (status == MMIPAL_SUCCESS)
213 {
214 printf("Execute cmd on AP 'iperf -c %s%%wlan0 -p %u -i 1 -V' for IPv6\n",
215 ip6_config.ip6_addr[0], args.local_port);
216 }
217}

260void app_init(void)
261{
262 printf("\n\nMorse Iperf Demo (Built " __DATE__ " " __TIME__ ")\n\n");
263
264 /* Initialize and connect to Wi-Fi, blocks till connected */
265 app_wlan_init();
266 app_wlan_start();
267
268 enum iperf_type iperf_mode = IPERF_TYPE;
269 char iperf_mode_str[32];
270 if (mmconfig_read_string("iperf.mode", iperf_mode_str, sizeof(iperf_mode_str)) > 0)
271 {
272 if (strncmp(iperf_mode_str, "udp_server", sizeof(iperf_mode_str)) == 0)
273 {
274 iperf_mode = IPERF_UDP_SERVER;
275 }
276 else if (strncmp(iperf_mode_str, "tcp_server", sizeof(iperf_mode_str)) == 0)
277 {
278 iperf_mode = IPERF_TCP_SERVER;
279 }
280 else if (strncmp(iperf_mode_str, "tcp_client", sizeof(iperf_mode_str)) == 0)
281 {
282 iperf_mode = IPERF_TCP_CLIENT;
283 }
284 else if (strncmp(iperf_mode_str, "udp_client", sizeof(iperf_mode_str)) == 0)
285 {
286 iperf_mode = IPERF_UDP_CLIENT;
287 }
288 else
289 {
290 printf("Unsupported iperf mode: %s\n", iperf_mode_str);
291 return;
292 }
293 }
294
295 switch (iperf_mode)
296 {
297 case IPERF_TCP_SERVER:
298 start_tcp_server();
299 break;
300
301 case IPERF_UDP_SERVER:
302 start_udp_server();
303 break;
304
305 case IPERF_UDP_CLIENT:
306 start_udp_client();
307 break;
308
309 case IPERF_TCP_CLIENT:
310 start_tcp_client();
311 break;
312 }
313}
