120static void execute_ping_request(int iteration)
121{
122 struct mmping_args args = MMPING_ARGS_DEFAULT;
123
124 set_debug_state(iteration == 0 ? DEBUG_STATE_PINGING_0 : DEBUG_STATE_PINGING_1);
125
126 /* Get the target IP */
127 strncpy(args.ping_target, DEFAULT_PING_REMOTE_IP, sizeof(args.ping_target));
128 mmconfig_read_string("ping.target", args.ping_target, sizeof(args.ping_target));
129
130 enum mmipal_status status = mmipal_get_local_addr(args.ping_src, args.ping_target);
131 if (status != MMIPAL_SUCCESS)
132 {
133 printf("failed to get local address for PING\n");
134 }
135 args.ping_count = DEFAULT_PING_COUNT;
136 mmconfig_read_uint32("ping.count", &args.ping_count);
137
138 args.ping_size = DEFAULT_PING_DATA_SIZE;
139 mmconfig_read_uint32("ping.target", &args.ping_size);
140
141 args.ping_interval_ms = DEFAULT_PING_INTERVAL_MS;
142 mmconfig_read_uint32("ping.interval", &args.ping_interval_ms);
143
144 mmping_start(&args);
145 printf("\nPing %s %lu(%lu) bytes of data.\n", args.ping_target, args.ping_size,
146 MMPING_ICMP_ECHO_HDR_LEN + args.ping_size);
147 mmosal_task_sleep(args.ping_interval_ms);
148
149 struct mmping_stats stats;
150 mmping_stats(&stats);
151 while (stats.ping_is_running)
152 {
153 mmosal_task_sleep(args.ping_interval_ms);
154 mmping_stats(&stats);
155 printf("(%s) packets transmitted/received = %lu/%lu, "
156 "round-trip min/avg/max = %lu/%lu/%lu ms\n",
157 stats.ping_receiver, stats.ping_total_count, stats.ping_recv_count,
158 stats.ping_min_time_ms, stats.ping_avg_time_ms, stats.ping_max_time_ms);
159 }
160 uint32_t loss = 0;
161 if (stats.ping_total_count == 0)
162 {
163 loss = 0;
164 }
165 else
166 {
167 loss = (1000 * (stats.ping_total_count - stats.ping_recv_count) * 100 /
168 stats.ping_total_count);
169 }
170
171 set_debug_state(iteration == 0 ? DEBUG_STATE_PING_0_DONE : DEBUG_STATE_PING_1_DONE);
172
173 printf("\n--- %s ping statistics ---\n%lu packets transmitted, %lu packets received, ",
174 stats.ping_receiver, stats.ping_total_count, stats.ping_recv_count);
175 printf("%lu.%03lu%% packet loss\nround-trip min/avg/max = %lu/%lu/%lu ms\n",
176 loss/1000, loss%1000, stats.ping_min_time_ms, stats.ping_avg_time_ms,
177 stats.ping_max_time_ms);
178}

185static void execute_wnm_sleep(uint32_t wnm_sleep_duration_ms)
186{
187 enum mmwlan_status status = MMWLAN_ERROR;
188
189 printf("\nEntering WNM sleep with chip power down disabled\n");
190 printf("Expected sleep time %lums.\n", wnm_sleep_duration_ms);
191 /* Delay to allow printf's to settle so we measure only idle current */
192 mmosal_task_sleep(100);
193
194 set_debug_state(DEBUG_STATE_WNM_SLEEP);
195
196 uint32_t timestamp = mmosal_get_time_ms();
197 status = mmwlan_set_wnm_sleep_enabled(true);
198 uint32_t wnm_sleep_enable_duration_ms = mmosal_get_time_ms() - timestamp;
199 if (status != MMWLAN_SUCCESS)
200 {
201 printf("Failed to enable WNM sleep\n");
202 return;
203 }
204
205 mmosal_task_sleep(wnm_sleep_duration_ms);
206
207 timestamp = mmosal_get_time_ms();
208
209 set_debug_state(DEBUG_STATE_EXITING_WNM_SLEEP);
210 status = mmwlan_set_wnm_sleep_enabled(false);
211 set_debug_state(DEBUG_STATE_WNM_SLEEP_DONE);
212
213 if (status == MMWLAN_SUCCESS)
214 {
215 printf("\nEnter WNM sleep took %lu ms.\n", wnm_sleep_enable_duration_ms);
216 printf("\nExit WNM sleep took %lu ms.\n", (mmosal_get_time_ms() - timestamp));
217 }
218 else
219 {
220 printf("Failed to disable WNM sleep\n");
221 }
222}

279void app_init(void)
280{
281 printf("\n\nMorse WNM Sleep Demo (Built " __DATE__ " " __TIME__ ")\n\n");
282
283 /* Initialize and connect to Wi-Fi, blocks till connected */
284 app_wlan_init();
285
286 set_debug_state(DEBUG_STATE_CONNECTING);
287
288 app_wlan_start();
289
290 set_debug_state(DEBUG_STATE_CONNECTED);
291
292 /* Delay to allow communications to settle so we measure only idle current */
293 mmosal_task_sleep(150);
294
295 uint32_t wnm_sleep_duration_ms = DEFAULT_WNM_SLEEP_DURATION_MS;
296 mmconfig_read_uint32("wlan.wnm_sleep_duration_ms", &wnm_sleep_duration_ms);
297
298 execute_ping_request(0);
299
300 /* The following demonstrates basic WNM Sleep functionality. The MM chip remains powered,
301 * but does not wake up to receive beacons. */
302 execute_wnm_sleep(wnm_sleep_duration_ms);
303
304 execute_ping_request(1);
305
306 /* The following demonstrates extended WNM Sleep API that allows the chip to be shutdown
307 * while in WNM Sleep. */
308 execute_wnm_sleep_ext(wnm_sleep_duration_ms);
309
310 /* Delay to allow communications to settle so we measure only idle current */
311 mmosal_task_sleep(150);
312
313 set_debug_state(DEBUG_STATE_TERMINATING);
314
315 /* Disconnect from Wi-Fi */
316 app_wlan_stop();
317}
