706static inline uint32_t mmosal_time_max(uint32_t a, uint32_t b)

1030static inline bool mmosal_safer_strcpy(char *dst, const char *src, size_t size)
1031{
1032 bool ret;
1033
1034 if (size == 0)
1035 {
1036 return true;
1037 }
1038
1039 strncpy(dst, src, size);
