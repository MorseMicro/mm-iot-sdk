759static inline bool mmosal_time_le(uint32_t a, uint32_t b)

780static inline uint32_t mmosal_time_max(uint32_t a, uint32_t b)
781{
782 if (mmosal_time_lt(a, b))
783 {
784 return b;
785 }
786 else
787 {
788 return a;
