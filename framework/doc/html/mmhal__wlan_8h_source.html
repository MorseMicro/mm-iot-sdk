607#define MMHAL_SDIO_ADDRESS_OFFSET (9)
609#define MMHAL_SDIO_ADDRESS_MAX ((1ul << 18) - 1)
610
612#define MMHAL_SDIO_COUNT_OFFSET (0)
614#define MMHAL_SDIO_COUNT_MAX ((1ul << 10) - 1)

617#define MMHAL_SDIO_CMD52_DATA_OFFSET (0)

630static inline uint32_t mmhal_make_cmd52_arg(enum mmhal_sdio_rw rw, enum mmhal_sdio_function fn,
631 uint32_t address, uint8_t write_data)
632{
633 uint32_t arg;
634
635 arg = rw | fn;
636 arg |= (address << MMHAL_SDIO_ADDRESS_OFFSET);
637 arg |= (write_data << MMHAL_SDIO_CMD52_DATA_OFFSET);
638 return arg;
639}
640
655static inline uint32_t mmhal_make_cmd53_arg(enum mmhal_sdio_rw rw, enum mmhal_sdio_function fn,
656 enum mmhal_sdio_mode mode, uint32_t address,
657 uint16_t count)
658{
659 uint32_t arg;
660
661 arg = rw | fn | MMHAL_SDIO_OPCODE_INC_ADDR | mode;
662 arg |= (address << MMHAL_SDIO_ADDRESS_OFFSET);
663 arg |= (count << MMHAL_SDIO_COUNT_OFFSET);
664 return arg;
665}
