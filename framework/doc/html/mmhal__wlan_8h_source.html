590#define MMHAL_SDIO_ADDRESS_OFFSET (9)

613static inline uint32_t mmhal_make_cmd52_arg(enum mmhal_sdio_rw rw, enum mmhal_sdio_function fn,
614 uint32_t address, uint8_t write_data)
615{
616 uint32_t arg;
617
618 arg = rw | fn;
619 arg |= (address << MMHAL_SDIO_ADDRESS_OFFSET);
620 arg |= (write_data << MMHAL_SDIO_CMD52_DATA_OFFSET);
621 return arg;
622}
623
638static inline uint32_t mmhal_make_cmd53_arg(enum mmhal_sdio_rw rw, enum mmhal_sdio_function fn,
639 enum mmhal_sdio_mode mode, uint32_t address,
640 uint16_t count)
641{
642 uint32_t arg;
643
644 arg = rw | fn | MMHAL_SDIO_OPCODE_INC_ADDR | mode;
645 arg |= (address << MMHAL_SDIO_ADDRESS_OFFSET);
646 arg |= (count << MMHAL_SDIO_COUNT_OFFSET);
647 return arg;
648}
